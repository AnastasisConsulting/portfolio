# engine/primes/core_prime_vector.py

from enum import Enum
from dataclasses import dataclass, field
from typing import List, Dict

class PrimeAxis(Enum):
    ALPHA = "Alpha"
    BETA = "Beta"
    GAMMA = "Gamma"

@dataclass
class PrimeVector:
    labels: List[str] = field(default_factory=lambda: ["Alpha", "Beta", "Gamma"])
    order: List[str] = field(default_factory=list)  # Sorted list of active prime labels
    focus: str = ""

    def reorder(self, new_order: List[str]):
        self.order = new_order

    def set_focus(self, prime: str):
        self.focus = prime

    def to_doi(self) -> Dict[str, str]:
        return {
            "First-Order": self.order[0],
            "Second-Order": self.order[1],
            "Third-Order": self.order[2]
        }
