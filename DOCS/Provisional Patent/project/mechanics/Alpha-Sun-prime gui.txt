On Prime Selection (via click or top-rank):
def on_prime_selected(prime_name: str):
    icon = QPixmap(f"assets/icons/{prime_name.lower()}.png")
    self.ui.label_prime_icon.setPixmap(icon)

    self.ui.text_prime_description.setText(load_prime_description(prime_name))

    sp = sub_primes[prime_name]
    self.ui.list_input.clear()
    self.ui.list_input.addItem(sp["input"])
    
    self.ui.list_expansion.clear()
    self.ui.list_expansion.addItem(sp["expansion"])
    
    self.ui.list_nuance.clear()
    self.ui.list_nuance.addItem(sp["nuance"])
